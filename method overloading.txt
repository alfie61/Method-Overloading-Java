class Camera {

    void click() {
        System.out.println("Photo captured");
    }

    void click(int seconds) {
        System.out.println("Video recorded for " + seconds + " seconds");
    }
}

public class Main {
    public static void main(String[] args) {

        Camera cam = new Camera();

        cam.click();        
        cam.click(10);      
    }
}
